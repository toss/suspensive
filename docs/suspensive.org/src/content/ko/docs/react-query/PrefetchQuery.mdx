# PrefetchQuery

React hookмқҳ м ңм•Ҫмқ„ н”јн•ҙ JSXмғҒм—җм„ңлҸ„ usePrefetchQueryлҘј мӮ¬мҡ©н•  мҲҳ мһҲкІҢ н•ҳлҠ” м»ҙнҸ¬л„ҢнҠёмһ…лӢҲлӢӨ.

```jsx /PrefetchQuery/
import { PrefetchQuery, useSuspenseQuery } from '@suspensive/react-query'

const PostsPage = () => {
  const { data: posts } = useSuspenseQuery({
    queryKey: ['posts'],
    queryFn: () => getPosts(),
  })

  return posts.map((post) => (
    <div key={post.id}>
      {/* рҹҡ« React hookмқҳ м ңм•ҪмңјлЎң usePrefetchQueryлҘј мӮ¬мҡ©н•ҳм§Җ лӘ»н•©лӢҲлӢӨ
      // usePrefetchQuery({
      //   queryKey: ['posts', post.id, 'comments'],
      //   queryFn: () => getPostComments(post.id),
      // })

      // вң… Post Comments нҺҳмқҙм§Җм—җ л“Өм–ҙк°Җкё° м „м—җ к°Ғк°Ғмқҳ post comments queryм—җ лҢҖн•ҙ usePrefetchQueryлҘј нҳём¶ңн•  мҲҳ мһҲмҠөлӢҲлӢӨ. */}
      <PrefetchQuery queryKey={['posts', post.id, 'comments']} queryFn={() => getPostComments(post.id)} />
      <h2>{post.title}</h2>
      <p>{post.description}</p>
      <Link to={`/posts/${post.id}/comments`}>See comments</Link>
    </div>
  ))
}
```
