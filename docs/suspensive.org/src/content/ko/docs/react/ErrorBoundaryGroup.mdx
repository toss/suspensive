# ErrorBoundaryGroup

мқҙ м»ҙнҸ¬л„ҢнҠёлЎң childrenмқё `<ErrorBoundary/>`л“Өмқ„ мүҪкІҢ кҙҖлҰ¬н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

мӨ‘мІ©лҗң `<ErrorBoundaryGroup/>`мқҳ childrenмқё `<ErrorBoundary/>`л“ӨлҸ„ мғҒмң„ `<ErrorBoundaryGroup.Consumer/>`мңјлЎңлҸ„ resetлҗ©лӢҲлӢӨ.

```jsx /ErrorBoundaryGroup/
import { ErrorBoundaryGroup, ErrorBoundary } from '@suspensive/react'

const Example = () => (
  <ErrorBoundaryGroup>
    {/* ErrorBoundaryGroupмқҳ childrenмқё ErrorBoundaryлҘј лӘЁл‘җ resetн•©лӢҲлӢӨ. мӨ‘мІ©лҗң ErrorBoundaryGroupлӮҙмқҳ ErrorBoundaryлҸ„ лӘЁл‘җ resetн•©лӢҲлӢӨ. */}
    <ErrorBoundaryGroup.Consumer>
      {(group) => <button onClick={group.reset}>Try again</button>}
    </ErrorBoundaryGroup.Consumer>
    <ErrorBoundary fallback={(props) => <>{props.error.message}</>}>
      <ErrorAfter2s />
    </ErrorBoundary>

    <ErrorBoundaryGroup>
      <ErrorBoundaryGroup.Consumer>
        {(group) => <button onClick={group.reset}>Try again</button>}
      </ErrorBoundaryGroup.Consumer>
      <ErrorBoundary fallback={(props) => <>{props.error.message}</>}>
        <ErrorAfter2s />
      </ErrorBoundary>
    </ErrorBoundaryGroup>
  </ErrorBoundaryGroup>
)
```

```tsx ErrorAfter2s.tsx
import { useState, useEffect } from 'react'

export const ErrorAfter2s = () => {
  const [asyncState, setAsyncState] = useState<
    { isError: true; error: Error } | { isError: false; error: null }
  >({
    isError: false,
    error: null,
  })

  useEffect(() => {
    setTimeout(() => {
      setAsyncState({
        isError: true,
        error: { status: 401, message: 'unauthorized' },
      })
    }, 2000)
  }, [])

  if (asyncState.isError) {
    throw asyncState.error
  }

  return <>No error</>
}
```
